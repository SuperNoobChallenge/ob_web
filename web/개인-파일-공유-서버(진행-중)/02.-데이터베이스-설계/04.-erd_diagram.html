// 1. 유저 및 권한 (Users & Permissions)

// --------------------------------------------------------------------------

Table users {

  user_id "bigint unsigned" [pk, increment, note: 'PK, 자동증가']

  file_permission_id "bigint unsigned" [not null, note: 'FK: 유저에게 할당된 루트 파일 권한']

  inviter_id "bigint unsigned" [null, note: 'FK: 나를 초대한 유저 (자기참조)']

  login_id varchar(255) [not null, unique, note: '로그인 아이디']

  password varchar(255) [not null, note: '암호화된 비밀번호']

  email varchar(50) [not null, unique, note: '이메일']

  deleted_at datetime [null, note: '비활성화 시기 (Soft Delete)']

  // Audit Columns

  created_at datetime [not null, default: now(), note: '생성 시각']

  updated_at datetime [not null, default: now(), note: '수정 시각']

  created_by "bigint unsigned" [not null, note: '생성자 ID (System=1)']

  updated_by "bigint unsigned" [not null, note: '수정자 ID']

}

// Refs for Users

Ref: users.user_id - users.inviter_id

Ref: file_permission_keys.file_permission_id - users.file_permission_id

Table user_permissions {

  user_id "bigint unsigned" [pk, note: 'PK & FK: users 테이블과 1:1']

  can_invite_person boolean [not null, default: false]

  can_share_file boolean [not null, default: false]

  can_store_personal boolean [not null, default: false]

  can_create_group boolean [not null, default: false]

  // Audit Columns

  created_at datetime [not null, default: now(), note: '생성 시각']

  updated_at datetime [not null, default: now(), note: '수정 시각']

  created_by "bigint unsigned" [not null, note: '생성자 ID (System=1)']

  updated_by "bigint unsigned" [not null, note: '수정자 ID']

}

// Refs for User Permissions

Ref: users.user_id - user_permissions.user_id

// 2. 파일 권한 및 용량 (File Permissions & Capacity)

// --------------------------------------------------------------------------

Table file_permission_keys {

  file_permission_id "bigint unsigned" [pk, increment]

  parent_permission_id "bigint unsigned" [null, note: 'FK: 상위 권한 키 (용량 상속)']

  owner_type varchar(10) [not null, note: 'USER, GROUP']

  total_capacity bigint [not null, default: 0, note: '총 할당 용량(Byte)']

  available_capacity bigint [not null, default: 0, note: '현재 사용 가능 용량(Byte)']

  deleted_at datetime [null]

  // Audit Columns

  created_at datetime [not null, default: now(), note: '생성 시각']

  updated_at datetime [not null, default: now(), note: '수정 시각']

  created_by "bigint unsigned" [not null, note: '생성자 ID (System=1)']

  updated_by "bigint unsigned" [not null, note: '수정자 ID']

}

// Refs for File Permission Keys

Ref:  file_permission_keys.file_permission_id - file_permission_keys.parent_permission_id

Table capacity_allocations {

  allocation_id "bigint unsigned" [pk, increment]

  granter_allocation_id "bigint unsigned" [null, note: 'FK: 용량을 준 상위 할당 ID']

  receiver_permission_id "bigint unsigned" [not null, note: 'FK: 용량을 받는 키']

  giver_permission_id "bigint unsigned" [null, note: 'FK: 용량을 주는 키 (System=NULL)']

  allocated_size bigint [not null, note: '할당되는 용량']

  expiration_date datetime [null, note: '만료일']

  allocation_type varchar(20) [not null, note: 'EVENT, GRANT, SYSTEM']

  description varchar(100) [null, note: '세부 내용']

  deleted_at datetime [null]

  // Audit Columns

  created_at datetime [not null, default: now(), note: '생성 시각']

  updated_at datetime [not null, default: now(), note: '수정 시각']

  created_by "bigint unsigned" [not null, note: '생성자 ID (System=1)']

  updated_by "bigint unsigned" [not null, note: '수정자 ID']

  indexes {

    (receiver_permission_id, deleted_at, expiration_date, allocated_size) [name: 'idx_capacity_allocations_active', note: '유효 용량 계산']

  }

}

// Refs for Capacity Allocations

// 3. 파일 시스템 (Virtual File System)

// --------------------------------------------------------------------------

Table virtual_directories {

  directory_id "bigint unsigned" [pk, increment]

  file_permission_id "bigint unsigned" [not null, note: 'FK: 소속 권한 키']

  parent_directory_id "bigint unsigned" [null, note: 'FK: 부모 디렉터리 (Root=NULL)']

  read_level int [not null, default: 0, note: '읽기 가능 레벨']

  write_level int [not null, default: 0, note: '쓰기 가능 레벨']

  name varchar(100) [not null, note: '디렉터리 이름']

  depth_level int [not null, default: 0, note: '디렉터리 깊이 (Root=0)']

  deleted_at datetime [null]

  // Audit Columns

  created_at datetime [not null, default: now(), note: '생성 시각']

  updated_at datetime [not null, default: now(), note: '수정 시각']

  created_by "bigint unsigned" [not null, note: '생성자 ID (System=1)']

  updated_by "bigint unsigned" [not null, note: '수정자 ID']

  indexes {

    (file_permission_id, parent_directory_id, deleted_at, name, directory_id) [name: 'idx_virtual_directories_parent_name', note: '부모 폴더 내 이름순 조회']

    (file_permission_id, parent_directory_id, deleted_at, updated_at, directory_id) [name: 'idx_virtual_directories_parent_updated', note: '부모 폴더 내 최신순 조회']

  }

}
