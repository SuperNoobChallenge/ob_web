Table batch_job_queues {

  batch_job_queue_id "bigint unsigned" [pk, increment, note: "배치 작업 대기열id (대리키)"]

  job_type varchar(50) [not null, note: "작업 유형(통계/집계 등)"]

  target_table varchar(64) [not null, note: "대상 테이블명(FK 미적용)"]

  target_id "bigint unsigned" [not null, note: "대상 테이블의 작업 대상 id"]

  status varchar(15) [not null, default: 'wait', note: "wait|in_progress|retry_wait|fail|success"]

  job_data json [not null, note: "작업 데이터(상세 파라미터)"]

  attempt_count int [not null, default: 0, note: "시도 횟수(현재까지 실행 시도된 횟수)"]

  max_attempts int [not null, note: "최대 실행 가능 횟수(재시도 허용 횟수 상한)"]

  next_run_at datetime [not null, note: "다음 실행 시각(이 시각 이후 실행 가능)"]

  started_at datetime [null, note: "가장 최근에 작업이 실행된 시각"]

  created_at datetime [not null, default: now(), note: "생성일"]

  finished_at datetime [null, note: "완료일(미완료 시 NULL)"]

  // 배치 작업은 updated_by(유저) 개념이 모호하므로

  // 수정일(updated_at)만 추가하거나, 제외해도 되지만 통일성을 위해 추가함

  updated_at datetime [not null, default: now()]

  created_by "bigint unsigned" [not null, note: 'System or User']

  updated_by "bigint unsigned" [not null, note: 'System']

  indexes {

    (status, next_run_at) [name: "idx_batch_job_queues_polling", note: "대기 작업 폴링(실행 가능 작업 조회)"]

  }

}
